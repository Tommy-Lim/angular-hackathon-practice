function minimumDifference(array){ if (array.length < 2) { return false } var smallest = array[1] - array[0]; for (i = 0; i < array.length; i++){ for (j = 0; j < array.length; j++){ if (i !== j) { var diff = array[i] - array[j]; diff = Math.abs(diff); smallest = Math.min(diff, smallest); } } } return smallest; };
